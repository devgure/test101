}

  id: String (PK)
  url: String
  isPrimary: Boolean
  facialScore: Float
}

  id: String (PK)
  latitude: Float
  longitude: Float
}

  id: String (PK)
  isSuperLike: Boolean
}

entity Match {
  id: String (PK)
  matchedAt: DateTime
}

entity ChatRoom {
  id: String (PK)
}

entity Message {
  id: String (PK)
  content: String
  isRead: Boolean
}

  id: String (PK)
  stripeId: String
  status: String
}
